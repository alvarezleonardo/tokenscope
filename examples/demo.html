Cómo usarlo

CLI
$ token-count analyze "tu prompt"

$ token-count analyze "tu prompt" \
    --providers anthropic,openai \
    --format json

$ token-count models --provider google

$ token-count refresh-prices

$ token-count web  # → localhost:8000
PYTHON
from token_contador import analyze, compare

r = analyze("tu texto")
print(r.cheapest)        # "gemini-2.0-flash"
print(r.largest_context) # "gemini-1.5-pro"

mr = r.by_model["gpt-4o"]
print(mr.tokens)         # 17
print(mr.cost.total_usd) # 0.005043

results = compare(textos, model_id="gpt-4o")
